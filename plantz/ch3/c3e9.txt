b) -1

Positive: 1 = 00000001
Bit flip: 11111110
+1      : 11111111h

So -1 = ff

c) -10

Positive: 10 = 00001010
Bit flip: 11110101
+1      : 11110110

So -10 = f6

d) 88 = 01011000 = 58

e) 127 = 01111111 = 7f

f) -16

Positive: 16 = 00010000
Bit flip: 11101111
+1      : 11110000 

So -16 = f0

g) -32

Positive: 32 = 00100000
Bit flip: 11011111
+1      : 11100000

So -32 = e0

h) -128

Positive: 128 = 10000000
Bit flip: 01111111
+1      : 10000000
          
So -128 = 80
